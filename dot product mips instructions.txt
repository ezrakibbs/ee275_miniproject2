        addu $r1 $r0 $r0 ;  result = 0
loop:   beq $r7 $r0 done ;  done looping?  
        lw $r2 0($r3) ;     load a elem  
        lw $r4 0($r5) ;     load b elem  
        mul $r2 $r2 $r4 ;   assume this is 1 instr. 
        addu $r1 $r1 $r2 ;  result += (*a) * (*b)  
        addiu $r3 $r3 #4  
        addiu $r5 $r5 #4  
        addiu $r7 $r7 #-1
        j loop  
done:   jr $r31
